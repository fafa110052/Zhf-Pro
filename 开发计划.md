## 3. 软件安装指南

### 3.1 Homebrew（Mac 包管理器）

如果你还没装 Homebrew，先安装它——后续所有软件都通过它安装：

```bash
/bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"
```

安装完成后验证：
```bash
brew --version
# 应输出：Homebrew 4.x.x
```

---

### 3.2 Node.js（后端 + 前端运行时）

**为什么需要**：Express 后端和 React 前端都运行在 Node.js 上，相当于 iOS 的"操作系统层"。

```bash
# 方式一：通过 nvm 安装（推荐，可以切换版本）
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.40.1/install.sh | bash
# 重启终端后
nvm install 20
nvm use 20

# 方式二：直接通过 Homebrew 安装
brew install node@20
```

**验证**：
```bash
node -v   # 应输出 v20.x.x
npm -v    # 应输出 10.x.x
```

> 💡 **iOS 类比**：安装 Node.js 就像安装 Xcode——这是你写代码和运行代码的基础环境。

---

### 3.3 VS Code（代码编辑器）

```bash
brew install --cask visual-studio-code
```

**必装插件**（打开 VS Code → 左侧 Extensions 图标 → 搜索安装）：

| 插件名 | 用途 | 类比 |
|--------|------|------|
| **ES7+ React/Redux/React-Native snippets** | React 代码片段 | Xcode 的 Code Snippets |
| **Prettier - Code formatter** | 自动格式化代码 | Xcode 的 Re-Indent |
| **Tailwind CSS IntelliSense** | TailwindCSS 自动补全 | — |
| **ESLint** | 代码质量检查 | SwiftLint |
| **REST Client** | API 测试（替代 Postman） | — |
| **SQLite Viewer** | 在 VS Code 内查看数据库 | Core Data 的 Data Model Editor |
| **微信小程序开发助手**（可选） | 小程序语法高亮 | — |

const categoryService = {

      .where('is_active', 1)
      .orderBy('sort_order', 'asc');

      if (!grouped[cat.type]) grouped[cat.type] = [];
      grouped[cat.type].push(cat);

    return grouped;
  },

  // 管理端：获取所有分类（含禁用）

    return db('categories')

      .orderBy('sort_order', 'asc');

    return db('categories').where('id', id).first();
  },

  // 更新分类
  async update(id, data) {
    await db('categories').where('id', id).update(data);

```js
const jwt = require('jsonwebtoken');
const config = require('../config');
const db = require('../db/connection');

// JWT Token 验证
const authenticate = async (req, res, next) => {
  try {
    const authHeader = req.headers.authorization;
    if (!authHeader || !authHeader.startsWith('Bearer ')) {
      return res.status(401).json({ error: { message: '未登录，请先登录' } });
    }

    const token = authHeader.split(' ')[1];
    const decoded = jwt.verify(token, config.jwtSecret);
    
    // 查询用户是否仍然存在且状态正常
    const user = await db('designers')
      .where('id', decoded.userId)
      .where('status', 'active')
      .first();
    
    if (!user) {
      return res.status(401).json({ error: { message: '用户不存在或已被禁用' } });
    }

    // 将用户信息挂载到请求对象，后续路由可用
    req.user = user;
    next();
  } catch (err) {
    if (err.name === 'TokenExpiredError') {
      return res.status(401).json({ error: { message: '登录已过期，请重新登录' } });
    }
    return res.status(401).json({ error: { message: '无效的认证令牌' } });
  }
};

// 角色权限检查
const requireRole = (...roles) => {
  return (req, res, next) => {
    if (!req.user) {
      return res.status(401).json({ error: { message: '未登录' } });
    }
    if (!roles.includes(req.user.role)) {
      return res.status(403).json({ error: { message: '没有操作权限' } });
    }
    next();
  };
};

module.exports = { authenticate, requireRole };
```

const express = require('express');
const router = express.Router();
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');
const config = require('../config');
const db = require('../db/connection');
const { authenticate, requireRole } = require('../middleware/auth');

// 管理员账号密码登录
router.post('/admin/login', async (req, res, next) => {
  try {
    const { username, password } = req.body;

    if (!username || !password) {
      return res.status(400).json({ error: { message: '请输入用户名和密码' } });
    }

    const admin = await db('designers')
      .where({ username, role: 'admin' })
      .first();

    if (!admin) {
      return res.status(401).json({ error: { message: '用户名或密码错误' } });
    }

    // 检查是否被锁定
    if (admin.locked_until && new Date(admin.locked_until) > new Date()) {
      const minutes = Math.ceil((new Date(admin.locked_until) - new Date()) / 60000);
      return res.status(423).json({
        error: { message: `账号已锁定，请 ${minutes} 分钟后再试` },
      });
    }

    // 验证密码
    const isValid = await bcrypt.compare(password, admin.password_hash);
    if (!isValid) {
      // 增加错误次数
      const attempts = (admin.login_attempts || 0) + 1;
      const updates = { login_attempts: attempts };

      // 5 次错误锁定 30 分钟
      if (attempts >= 5) {
        updates.locked_until = new Date(Date.now() + 30 * 60 * 1000).toISOString();
        updates.login_attempts = 0;
      }

      await db('designers').where('id', admin.id).update(updates);

      return res.status(401).json({ error: { message: '用户名或密码错误' } });
    }

    // 登录成功，重置错误计数
    await db('designers').where('id', admin.id).update({
      login_attempts: 0,
      locked_until: null,
    });

    // 生成 JWT
    const token = jwt.sign(
      { userId: admin.id, role: admin.role },
      config.jwtSecret,
      { expiresIn: config.jwtExpiresIn }
    );

    res.json({
      success: true,
      data: {
        token,
        user: {
          id: admin.id,
          name: admin.name,
          role: admin.role,
          avatar_url: admin.avatar_url,
        },
      },
    });
  } catch (err) {
    next(err);
  }
});

**步骤 3：封装 API 请求**

```js
const BASE_URL = 'http://localhost:3000/api/v1';  // 开发环境

const request = (options) => {
  const token = wx.getStorageSync('token');
  
  return new Promise((resolve, reject) => {
    wx.request({
      url: BASE_URL + options.url,
      method: options.method || 'GET',
      data: options.data,
      header: {
        'Content-Type': 'application/json',
        ...(token ? { 'Authorization': `Bearer ${token}` } : {}),
      },
      success(res) {
        if (res.statusCode >= 200 && res.statusCode < 300) {
          resolve(res.data);
        } else if (res.statusCode === 401) {
          wx.removeStorageSync('token');
          wx.navigateTo({ url: '/pages/designer-login/designer-login' });
          reject(res.data);
        } else {
          reject(res.data);
        }
      },
      fail(err) {
        wx.showToast({ title: '网络请求失败', icon: 'error' });
        reject(err);
      },
    });
  });
};

module.exports = { request, BASE_URL };
```

## 12. 常见问题 FAQ

### Q1: 我是 iOS 开发者，没用过 JavaScript，需要先系统学习吗？

**答**：不需要。如果你熟悉 Swift，JavaScript 的语法非常相似。主要区别：

| Swift | JavaScript |
|-------|-----------|
| `let name: String = "hello"` | `const name = "hello"` |
| `var count: Int = 0` | `let count = 0` |
| `func foo() -> String {}` | `function foo() {}` |
| `if let value = optional {}` | `if (value) {}` |
| `[String: Any]` | `{}` (Object) |
| `[String]` | `[]` (Array) |
| `try await` | `async/await` (语法相同) |
| `guard let` | `if (!value) return` |

每天开发中遇到不熟悉的语法，直接问 Claude 就行。

### Q2: npm install 很慢怎么办？

```bash
# 设置国内镜像
npm config set registry https://registry.npmmirror.com
```

### Q3: 微信小程序请求不到后台 API 怎么办？

1. 检查后端是否启动：`curl http://localhost:3000/api/health`
2. 小程序开发工具中：详情 → 本地设置 → 勾选"不校验合法域名"
3. 真机预览：需要在微信公众平台配置合法域名（需要 HTTPS）

### Q4: React 的 JSX 语法看不懂？

JSX 就是 HTML 写在 JavaScript 里。SwiftUI 开发者应该很熟悉：

```swift
// SwiftUI
VStack {
    Text("Hello")
    Image("icon")
}
```

```jsx
// React JSX
<div className="flex flex-col">
    <span>Hello</span>
    <img src="icon.png" />
</div>
```

### Q5: 数据库文件被锁了怎么办？

```bash
# SQLite 单写锁，确保只有一个进程在访问
lsof /Users/lyf/Desktop/ZHFPro/server/data/database.sqlite
# 如果有其他进程，kill 掉
```

### Q6: 想回滚数据库到初始状态？

```bash
cd /Users/lyf/Desktop/ZHFPro/server
npm run migrate:reset
# 这会：删表 → 重建 → 插入种子数据
```

> ⚠️ 这会删除所有数据！仅开发环境使用。
